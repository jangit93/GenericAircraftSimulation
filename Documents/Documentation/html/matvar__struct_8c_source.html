45 Mat_VarCreateStruct(const char *name,int rank,size_t *dims,const char **fields,
46  unsigned nfields)
47 {
48  int i, nmemb = 1;
49  matvar_t *matvar;
50 
51  if ( NULL == dims )
52  return NULL;
53 
54  matvar = Mat_VarCalloc();
55  if ( NULL == matvar )
56  return NULL;
57 
58  matvar->compression = MAT_COMPRESSION_NONE;
59  if ( NULL != name )
60  matvar->name = strdup(name);
61  matvar->rank = rank;
62  matvar->dims = (size_t*)malloc(matvar->rank*sizeof(*matvar->dims));
63  for ( i = 0; i < matvar->rank; i++ ) {
64  matvar->dims[i] = dims[i];
65  nmemb *= dims[i];
66  }
67  matvar->class_type = MAT_C_STRUCT;
68  matvar->data_type = MAT_T_STRUCT;
69 
70  matvar->data_size = sizeof(matvar_t *);
71 
72  if ( nfields ) {
73  matvar->internal->num_fields = nfields;
74  matvar->internal->fieldnames =
75  (char**)malloc(nfields*sizeof(*matvar->internal->fieldnames));
76  if ( NULL == matvar->internal->fieldnames ) {
77  Mat_VarFree(matvar);
78  matvar = NULL;
79  } else {
80  for ( i = 0; i < nfields; i++ ) {
81  if ( NULL == fields[i] ) {
82  Mat_VarFree(matvar);
83  matvar = NULL;
84  break;
85  } else {
86  matvar->internal->fieldnames[i] = strdup(fields[i]);
87  }
88  }
89  }
90  if ( NULL != matvar && nmemb > 0 ) {
91  matvar_t **field_vars;
92  matvar->nbytes = nmemb*nfields*matvar->data_size;
93  matvar->data = malloc(matvar->nbytes);
94  field_vars = (matvar_t**)matvar->data;
95  for ( i = 0; i < nfields*nmemb; i++ )
96  field_vars[i] = NULL;
97  }
98  }
99 
100  return matvar;
101 }

113 int
114 Mat_VarAddStructField(matvar_t *matvar,const char *fieldname)
115 {
116  int i, f, nfields, nmemb, cnt = 0;
117  matvar_t **new_data, **old_data;
118  char **fieldnames;
119 
120  if ( matvar == NULL || fieldname == NULL )
121  return -1;
122  nmemb = 1;
123  for ( i = 0; i < matvar->rank; i++ )
124  nmemb *= matvar->dims[i];
125 
126  nfields = matvar->internal->num_fields+1;
127  matvar->internal->num_fields = nfields;
128  fieldnames = (char**)realloc(matvar->internal->fieldnames,
129  nfields*sizeof(*matvar->internal->fieldnames));
130  if ( NULL == fieldnames )
131  return -1;
132  matvar->internal->fieldnames = fieldnames;
133  matvar->internal->fieldnames[nfields-1] = strdup(fieldname);
134 
135  new_data = (matvar_t**)malloc(nfields*nmemb*sizeof(*new_data));
136  if ( new_data == NULL )
137  return -1;
138 
139  old_data = (matvar_t**)matvar->data;
140  for ( i = 0; i < nmemb; i++ ) {
141  for ( f = 0; f < nfields-1; f++ )
142  new_data[cnt++] = old_data[i*(nfields-1)+f];
143  new_data[cnt++] = NULL;
144  }
145 
146  free(matvar->data);
147  matvar->data = new_data;
148  matvar->nbytes = nfields*nmemb*sizeof(*new_data);
149 
150  return 0;
151 }

337 matvar_t *
338 Mat_VarGetStructs(matvar_t *matvar,int *start,int *stride,int *edge,
339  int copy_fields)
340 {
341  size_t i,j,N,I,nfields,field,idx[10] = {0,},cnt[10] = {0,},dimp[10] = {0,};
342  matvar_t **fields, *struct_slab;
343 
344  if ( (matvar == NULL) || (start == NULL) || (stride == NULL) ||
345  (edge == NULL) ) {
346  return NULL;
347  } else if ( matvar->rank > 9 ) {
348  return NULL;
349  } else if ( matvar->class_type != MAT_C_STRUCT ) {
350  return NULL;
351  }
352 
353  struct_slab = Mat_VarDuplicate(matvar,0);
354  if ( !copy_fields )
355  struct_slab->mem_conserve = 1;
356 
357  nfields = matvar->internal->num_fields;
358 
359  dimp[0] = matvar->dims[0];
360  N = edge[0];
361  I = start[0];
362  struct_slab->dims[0] = edge[0];
363  idx[0] = start[0];
364  for ( i = 1; i < matvar->rank; i++ ) {
365  idx[i] = start[i];
366  dimp[i] = dimp[i-1]*matvar->dims[i];
367  N *= edge[i];
368  I += start[i]*dimp[i-1];
369  struct_slab->dims[i] = edge[i];
370  }
371  I *= nfields;
372  struct_slab->nbytes = N*nfields*sizeof(matvar_t *);
373  struct_slab->data = malloc(struct_slab->nbytes);
374  if ( struct_slab->data == NULL ) {
375  Mat_VarFree(struct_slab);
376  return NULL;
377  }
378  fields = (matvar_t**)struct_slab->data;
379  for ( i = 0; i < N; i+=edge[0] ) {
380  for ( j = 0; j < edge[0]; j++ ) {
381  for ( field = 0; field < nfields; field++ ) {
382  if ( copy_fields )
383  fields[(i+j)*nfields+field] =
384  Mat_VarDuplicate(*((matvar_t **)matvar->data + I),1);
385  else
386  fields[(i+j)*nfields+field] =
387  *((matvar_t **)matvar->data + I);
388  I++;
389  }
390  I += (stride[0]-1)*nfields;
391  }
392  idx[0] = start[0];
393  I = idx[0];
394  cnt[1]++;
395  idx[1] += stride[1];
396  for ( j = 1; j < matvar->rank; j++ ) {
397  if ( cnt[j] == edge[j] ) {
398  cnt[j] = 0;
399  idx[j] = start[j];
400  cnt[j+1]++;
401  idx[j+1] += stride[j+1];
402  }
403  I += idx[j]*dimp[j-1];
404  }
405  I *= nfields;
406  }
407  return struct_slab;
408 }
