78 Mat_VarGetCells(matvar_t *matvar,int *start,int *stride,int *edge)
79 {
80  int i, j, N, I;
81  size_t idx[10] = {0,}, cnt[10] = {0,}, dimp[10] = {0,};
82  matvar_t **cells;
83 
84  if ( (matvar == NULL) || (start == NULL) || (stride == NULL) ||
85  (edge == NULL) ) {
86  return NULL;
87  } else if ( matvar->rank > 9 ) {
88  return NULL;
89  }
90 
91  dimp[0] = matvar->dims[0];
92  N = edge[0];
93  I = start[0];
94  idx[0] = start[0];
95  for ( i = 1; i < matvar->rank; i++ ) {
96  idx[i] = start[i];
97  dimp[i] = dimp[i-1]*matvar->dims[i];
98  N *= edge[i];
99  I += start[i]*dimp[i-1];
100  }
101  cells = (matvar_t**)malloc(N*sizeof(matvar_t *));
102  for ( i = 0; i < N; i+=edge[0] ) {
103  for ( j = 0; j < edge[0]; j++ ) {
104  cells[i+j] = *((matvar_t **)matvar->data + I);
105  I += stride[0];
106  }
107  idx[0] = start[0];
108  I = idx[0];
109  cnt[1]++;
110  idx[1] += stride[1];
111  for ( j = 1; j < matvar->rank; j++ ) {
112  if ( cnt[j] == edge[j] ) {
113  cnt[j] = 0;
114  idx[j] = start[j];
115  cnt[j+1]++;
116  idx[j+1] += stride[j+1];
117  }
118  I += idx[j]*dimp[j-1];
119  }
120  }
121  return cells;
122 }
