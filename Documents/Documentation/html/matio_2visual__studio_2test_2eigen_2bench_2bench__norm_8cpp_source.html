34 {
35  typedef typename T::Scalar Scalar;
36  int n = v.size();
37  Scalar scale = 0;
38  Scalar ssq = 1;
39  for (int i=0;i<n;++i)
40  {
41  Scalar ax = std::abs(v.coeff(i));
42  if (scale >= ax)
43  {
44  ssq += numext::abs2(ax/scale);
45  }
46  else
47  {
48  ssq = Scalar(1) + ssq * numext::abs2(scale/ax);
49  scale = ax;
50  }
51  }
52  return scale * std::sqrt(ssq);
53 }
