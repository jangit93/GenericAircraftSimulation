1 !************************************************************
2 !
3 ! This example shows how to create "compact-or-indexed"
4 ! format groups, new to 1.8. This example also illustrates
5 ! the space savings of compact groups by creating 2 files
6 ! which are identical except for the group format, and
7 ! displaying the file size of each. Both files have one
8 ! empty group in the root group.
9 !
10 ! This file is intended for use with HDF5 Library verion 1.8
11 !
12 !************************************************************
13 
14 PROGRAM main
15 
16  USE hdf5
17  IMPLICIT NONE
18 
19  CHARACTER(LEN=18), PARAMETER :: filename1 = "h5ex_g_compact1.h5"
20  CHARACTER(LEN=18), PARAMETER :: filename2 = "h5ex_g_compact2.h5"
21  CHARACTER(LEN=2) , PARAMETER :: groupname = "G1"
22 
23  INTEGER(HID_T) :: file, group, fapl ! handles
24  INTEGER :: hdferr
25  INTEGER :: storage_type ! Type of storage for links in group:
26  ! H5G_STORAGE_TYPE_COMPACT: Compact storage
27  ! H5G_STORAGE_TYPE_DENSE: Indexed storage
28  ! H5G_STORAGE_TYPE_SYMBOL_TABLE: Symbol tables
29  INTEGER :: nlinks ! Number of links in group
30  INTEGER :: max_corder ! Current maximum creation order value for group
31  INTEGER(HSIZE_T) :: size
32  !
33  ! Initialize FORTRAN interface.
34  !
35  CALL h5open_f(hdferr)
36  !
37  ! Create file 1. This file will use original format groups.
38  !
39  CALL h5fcreate_f(filename1, h5f_acc_trunc_f, file, hdferr)
40  CALL h5gcreate_f(file, groupname, group, hdferr)
41  !
42  ! Obtain the group info and print the group storage type.
43  !
44  CALL h5gget_info_f(group, storage_type, nlinks, max_corder, hdferr)
45 
46  WRITE(*,'("Group storage type for ",A," is: ")', advance='NO' ) filename1
47 
48  IF(storage_type.EQ.h5g_storage_type_compact_f)THEN
49  WRITE(*,'("H5G_STORAGE_TYPE_COMPACT_F")') ! New compact format
50  ELSE IF(storage_type.EQ.h5g_storage_type_dense_f)THEN
51  WRITE(*,'("H5G_STORAGE_TYPE_DENSE_F")') ! New dense (indexed) format
52  ELSE IF(storage_type.EQ.h5g_storage_type_symbol_table_f)THEN
53  WRITE(*,'("H5G_STORAGE_TYPE_SYMBOL_TABLE_F")') ! Original format
54  ENDIF
55  !
56  ! Close and re-open file. Needed to get the correct file size.
57  !
58  CALL h5gclose_f(group, hdferr)
59  CALL h5fclose_f(file , hdferr)
60  CALL h5fopen_f(filename1, h5f_acc_rdonly_f, file, hdferr)
61  !
62  ! Obtain and print the file size.
63  !
64  CALL h5fget_filesize_f(file, size, hdferr)
65  WRITE(*,'("File size for ",A," is: ",i6," bytes",/)') filename1, size
66  !
67  ! Close filename1
68  !
69  CALL h5fclose_f(file, hdferr)
70  !
71  ! Set file access property list to allow the latest file format.
72  ! This will allow the library to create new compact format groups.
73  !
74  CALL h5pcreate_f (h5p_file_access_f, fapl, hdferr)
75  CALL h5pset_libver_bounds_f (fapl, h5f_libver_latest_f, h5f_libver_latest_f, hdferr)
76  !
77  ! Create file 2 using the new file access property list.
78  !
79  CALL h5fcreate_f(filename2, h5f_acc_trunc_f, file, hdferr, access_prp=fapl)
80  CALL h5gcreate_f(file, groupname, group, hdferr)
81  !
82  ! Obtain the group info and print the group storage type.
83  !
84  CALL h5gget_info_f(group, storage_type, nlinks, max_corder, hdferr)
85 
86  WRITE(*,'("Group storage type for ",A," is: ")', advance='NO' ) filename2
87 
88  IF(storage_type.EQ.h5g_storage_type_compact_f)THEN
89  WRITE(*,'("H5G_STORAGE_TYPE_COMPACT_F")') ! New compact format
90  ELSE IF(storage_type.EQ.h5g_storage_type_dense_f)THEN
91  WRITE(*,'("H5G_STORAGE_TYPE_DENSE_F")') ! New dense (indexed) format
92  ELSE IF(storage_type.EQ.h5g_storage_type_symbol_table_f)THEN
93  WRITE(*,'("H5G_STORAGE_TYPE_SYMBOL_TABLE_F")') ! Original format
94  ELSE IF(storage_type.EQ.h5g_storage_type_unknown_f)THEN
95  WRITE(*,'("H5G_STORAGE_TYPE_UNKNOWN_F")') ! Unknown format
96  ENDIF
97  !
98  ! Close and re-open file. Needed to get the correct file size.
99  !
100  CALL h5gclose_f(group, hdferr)
101  CALL h5fclose_f(file , hdferr)
102  CALL h5fopen_f(filename2, h5f_acc_rdonly_f, file, hdferr, fapl)
103  !
104  ! Obtain and print the file size.
105  !
106  CALL h5fget_filesize_f(file, size, hdferr)
107  WRITE(*,'("File size for ",A," is: ",i6,"bytes",/)') filename2, size
108  !
109  ! Close and release resources.
110  !
111  CALL h5pclose_f(fapl, hdferr)
112  CALL h5fclose_f(file, hdferr)
113 
114 END PROGRAM main
