66 void DATCOMAerodymamic::updateAerodynamic(Float64 FlightTime,
67  AtmosphereStruct & AtmoData,
68  AerodynamicStruct & AeroData,
69  AirframeStruct & AirframeData,
70  ThrustStruct & ThrustData)
71 {
72  relVelNED = AirframeData.velNED;
73 
74  absVel = sqrt(relVelNED(0)*relVelNED(0)+ relVelNED(1)*relVelNED(1)+ relVelNED(2)*relVelNED(2));
75  rho = AtmoData.rho;
76  SpeedOfSound = AtmoData.speedOfSound;
77 
78  Ma = absVel / SpeedOfSound;
79  qbar = 0.5*rho*absVel*absVel;
80 
81 
82  velBody = AirframeData.matNEDToBody*AirframeData.velNED;
83  AirframeData.velBody = velBody;
84 
85  Alpha = atan2(velBody(2), velBody(0)) * 180 / PI;
86  Beta = atan2(velBody(1), sqrt(velBody(0)*velBody(0)+ velBody(1)*velBody(1)+ velBody(2)*velBody(2))) * 180 / PI;
87 
88  ETA = AirframeData.Eta * 180 / PI;
89  XI = AirframeData.Xi * 180 / PI;
90  ZETA = AirframeData.Zeta * 180 / PI;
91 
92  p = AirframeData.rotRatesBody(0) * b / 2 * absVel;
93  q = AirframeData.rotRatesBody(1) * l_nu*absVel;
94  r = AirframeData.rotRatesBody(2)* b / 2 * absVel;
95 
96 
97  if (Ma >= 0.9)
98  {
99  Ma = 0.9;
100  }
101 
102  c_a = Interpolation.linearInterpolation2D(AoA, Mach, CA, Alpha, Ma);
103  c_aEta = Interpolation.linearInterpolation2D(Eta, Mach, CAeta, ETA, Ma);
104  c_w = Interpolation.linearInterpolation2D(AoA, Mach, CW, Alpha, Ma);
105  c_weta = Interpolation.linearInterpolation2D(Eta, Mach, CWeta, ETA, Ma);
106  c_m = Interpolation.linearInterpolation2D(AoA, Mach, CM, Alpha, Ma);
107  c_meta = Interpolation.linearInterpolation2D(Eta, Mach, CMeta, ETA, Ma);
108  c_mq = Interpolation.linearInterpolation1D(Mach, CMq,Ma);
109  c_aq = Interpolation.linearInterpolation1D(Mach, CAq, Ma);
110 
111  c_y = Interpolation.linearInterpolation2D(AoS, Mach, CQ, Beta, Ma);
112  c_yp = Interpolation.linearInterpolation2D(AoA, Mach, CQp, Alpha, Ma);
113  c_yzeta = CQzeta;
114  c_yxi = CQzeta;
115 
116  c_l = Interpolation.linearInterpolation2D(AoS, Mach, CL, Beta, Ma);
117  c_lxi = Interpolation.linearInterpolation2D(Xi, Mach, CLxi, XI, Ma);
118  c_lp = Interpolation.linearInterpolation2D(AoA, Mach, CLp, Alpha, Ma);
119  c_lr = Interpolation.linearInterpolation2D(AoA, Mach, CLr, Alpha, Ma);
120  c_lzeta = CLzeta;
121 
122  c_n = Interpolation.linearInterpolation2D(AoS, Mach, CN, Beta, Ma);
123  c_nxi = Interpolation.linearInterpolation2D(Xi, Mach, CNxi, XI, Ma);
124  c_np = Interpolation.linearInterpolation2D(AoA, Mach, CNp, Alpha, Ma);
125  c_nr = Interpolation.linearInterpolation2D(AoA, Mach, CNr, Alpha, Ma);
126  c_nzeta = CNzeta;
127 
128  alpha = Alpha * PI / 180;
129  beta = Beta * PI / 180;
130  eta = ETA * PI / 180;
131  xi = XI * PI / 180;
132  zeta = ZETA * PI / 180;
133 
134  C_A = c_a + c_aq * q + c_aEta *eta;
135 
136  C_W = c_w + c_weta * eta;
137 
138  C_Q = c_y + c_yp*p + c_yzeta*zeta + c_yxi*xi; //Fehler in Aeromodell BA!!!!!
139 
140  C_L = c_l + c_lp * p + c_lr * r + c_lxi * xi + c_lzeta * zeta;
141 
142  C_M = c_m + c_mq * q + c_meta * eta;
143 
144  C_N = c_n + c_np * p + c_nr *r + c_nxi * xi + c_nzeta*zeta;
145 
146  C_X = -C_W * cos(alpha) * cos(beta) - C_Q *cos(alpha) * sin(beta) + C_A * sin(alpha);
147 
148  C_Y = C_Q * cos(beta) - C_W*sin(beta);
149 
150  C_Z = -C_W * sin(alpha) * cos(beta) - C_Q *sin(alpha) * sin(beta) - C_A * cos(alpha);
151 
152  // Guidanc daten noch ergänzen
153  Float64 alpha0 = 0;
154  ca_alpha = Interpolation.linearInterpolation2D(AoA, Mach, CAalpha, Alpha, Ma);
155  ca_0 = Interpolation.linearInterpolation2D(AoA, Mach, CA, alpha0, Ma);
156  c_zdalpha = -C_W - ca_alpha;
157 
158  // Kräfte und Momente

161  AeroData.AeroForces(2) = C_Z * qbar*S;
162 
163  AeroData.AeroMoments(0) = C_L * qbar*S * (b / 2);
164  AeroData.AeroMoments(1) = C_M * qbar*S * l_nu;
165  AeroData.AeroMoments(2) = C_N * qbar*S * (b / 2);
166

168  AeroData.C_zdalpha = c_zdalpha;
169  AeroData.C_A0 = ca_0;
170 
171  AeroData.CA = C_A;
