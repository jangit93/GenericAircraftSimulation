1 !************************************************************
2 !
3 ! This example shows how to read and write opaque datatypes
4 ! to a dataset. The program first writes opaque data to a
5 ! dataset with a dataspace of DIM0, then closes the file.
6 ! Next, it reopens the file, reads back the data, and
7 ! outputs it to the screen.
8 !
9 ! This file is intended for use with HDF5 Library version 1.8
10 ! with --enable-fortran2003
11 !
12 !************************************************************
13 PROGRAM main
14 
15  USE hdf5
16  USE iso_c_binding
17 
18  IMPLICIT NONE
19  CHARACTER(LEN=20), PARAMETER :: filename = "h5ex_t_opaque_F03.h5"
20  CHARACTER(LEN=3) , PARAMETER :: dataset = "DS1"
21  INTEGER , PARAMETER :: dim0 = 4
22  INTEGER(SIZE_T) , PARAMETER :: size = 7
23  INTEGER(HID_T) :: file, space, dtype, dset ! Handles
24  INTEGER :: hdferr
25  INTEGER(size_t) :: len
26  INTEGER(hsize_t), DIMENSION(1:1) :: dims = (/dim0/)
27 
28  CHARACTER(LEN=size), DIMENSION(1:dim0), TARGET :: wdata ! Write buffer
29  CHARACTER(LEN=size), DIMENSION(:), ALLOCATABLE, TARGET :: rdata ! Read buffer
30  CHARACTER(LEN=size-1) :: str = "OPAQUE"
31  CHARACTER(LEN=80) :: tag
32  INTEGER :: taglen
33  INTEGER(HSIZE_T), DIMENSION(1:1) :: maxdims
34  INTEGER :: i
35  CHARACTER(LEN=1) :: ichr
36  TYPE(c_ptr) :: f_ptr
37  !
38  ! Initialize FORTRAN interface.
39  !
40  CALL h5open_f(hdferr)
41  !
42  ! Initialize data.
43  !
44  DO i = 1, dim0
45  WRITE(ichr,'(I1)') i-1
46  wdata(i) = str//ichr
47  ENDDO
48  !
49  ! Create a new file using the default properties.
50  !
51  CALL h5fcreate_f(filename, h5f_acc_trunc_f, file, hdferr)
52  !
53  ! Create opaque datatype and set the tag to something appropriate.
54  ! For this example we will write and view the data as a character
55  ! array.
56  !
57  CALL h5tcreate_f(h5t_opaque_f, size, dtype, hdferr)
58  CALL h5tset_tag_f(dtype,"Character array",hdferr)
59  !
60  ! Create dataspace. Setting maximum size to be the current size.
61  !
62  CALL h5screate_simple_f(1, dims, space, hdferr)
63  !
64  ! Create the dataset and write the opaque data to it.
65  !
66  CALL h5dcreate_f(file, dataset, dtype, space, dset, hdferr)
67  f_ptr = c_loc(wdata(1))
68  CALL h5dwrite_f(dset, dtype, f_ptr, hdferr)
69  !
70  ! Close and release resources.
71  !
72  CALL h5dclose_f(dset, hdferr)
73  CALL h5sclose_f(space, hdferr)
74  CALL h5tclose_f(dtype, hdferr)
75  CALL h5fclose_f(file, hdferr)
76  !
77  ! Now we begin the read section of this example.
78  !
79  ! Open file and dataset.
80  !
81  CALL h5fopen_f(filename, h5f_acc_rdonly_f, file, hdferr)
82  CALL h5dopen_f(file, dataset, dset, hdferr)
83  !
84  ! Get datatype and properties for the datatype.
85  !
86  CALL h5dget_type_f(dset, dtype, hdferr)
87  CALL h5tget_size_f(dtype, len, hdferr)
88  CALL h5tget_tag_f(dtype, tag, taglen, hdferr)
89  !
90  ! Get dataspace and allocate memory for read buffer.
91  !
92  CALL h5dget_space_f(dset, space, hdferr)
93  CALL h5sget_simple_extent_dims_f(space, dims, maxdims, hdferr)
94  ALLOCATE(rdata(1:dims(1)))
95  !
96  ! Read the data.
97  !
98  f_ptr = c_loc(rdata(1))
99  CALL h5dread_f(dset, dtype, f_ptr, hdferr)
100  !
101  ! Output the data to the screen.
102  !
103  WRITE(*,'("Datatype tag for ",A," is: ",A)') dataset, '"'//tag(1:taglen)//'"'
104  !
105  DO i = 1, dims(1)
106  WRITE(*,'(A,"[",i1,"]: ",A)') dataset,i-1,rdata(i)
107  ENDDO
108  !
109  ! Close and release resources.
110  !
111  DEALLOCATE(rdata)
112  CALL h5dclose_f(dset, hdferr)
113  CALL h5sclose_f(space, hdferr)
114  CALL h5tclose_f(dtype, hdferr)
115  CALL h5fclose_f(file, hdferr)
116  !
117 END PROGRAM main
