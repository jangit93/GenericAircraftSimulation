23  H5G_info_t ginfo;
24  ssize_t size; /* Size of name */

26  char *name; /* Output buffer */

28  /*
29  * Create a new file using the default properties.
30  */
31  file = H5Fcreate (FILE, H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
32 
33  /*

38  */
39  gcpl = H5Pcreate (H5P_GROUP_CREATE);

45  */

47 
48  /*

50  * by creation order. Note that these groups do not have to have
51  * the creation order tracking property set.
52  */
53  subgroup = H5Gcreate (group, "H", H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
54  status = H5Gclose (subgroup);

60  status = H5Gclose (subgroup);
61

64  */

66

69  * (H5_INDEX_NAME).
70  */

72  for (i=0; i<ginfo.nlinks; i++) {
73

76  */
77  size = 1 + H5Lget_name_by_idx (group, ".", H5_INDEX_NAME, H5_ITER_INC,
78  i, NULL, 0, H5P_DEFAULT);
79 
80  /*
