1 ! ************************************************************
2 !
3 ! This example shows how to read and write data to a compact
4 ! dataset. The program first writes integers to a compact
5 ! dataset with dataspace dimensions of DIM0xDIM1, then
6 ! closes the file. Next, it reopens the file, reads back
7 ! the data, and outputs it to the screen.
8 !
9 ! This file is intended for use with HDF5 Library verion 1.8
10 !
11 ! ************************************************************
12 
13 PROGRAM main
14 
15  USE hdf5
16 
17  IMPLICIT NONE
18 
19  CHARACTER(LEN=17), PARAMETER :: filename = "h5ex_d_compact.h5"
20  CHARACTER(LEN=3) , PARAMETER :: dataset = "DS1"
21  INTEGER , PARAMETER :: dim0 = 4
22  INTEGER , PARAMETER :: dim1 = 7
23 
24  INTEGER :: hdferr
25  INTEGER :: layout
26  INTEGER(HID_T) :: file, space, dset, dcpl ! Handles
27  INTEGER(HSIZE_T), DIMENSION(1:2) :: dims = (/dim0, dim1/)
28 
29  INTEGER, DIMENSION(1:dim0, 1:dim1) :: wdata, & ! Write buffer
30  rdata ! Read buffer
31  INTEGER :: i, j
32  !
33  ! Initialize FORTRAN interface.
34  !
35  CALL h5open_f(hdferr)
36  !
37  ! Initialize data.
38  !
39  DO i = 1, dim0
40  DO j = 1, dim1
41  wdata(i,j) = (i-1)*(j-1)-(j-1)
42  ENDDO
43  ENDDO
44  !
45  ! Create a new file using the default properties.
46  !
47  CALL h5fcreate_f(filename, h5f_acc_trunc_f, file, hdferr)
48  !
49  ! Create dataspace. Setting maximum size to NULL sets the maximum
50  ! size to be the current size.
51  !
52  CALL h5screate_simple_f(2, dims, space, hdferr)
53  !
54  ! Create the dataset creation property list, set the layout to
55  ! compact.
56  !
57  CALL h5pcreate_f(h5p_dataset_create_f, dcpl, hdferr)
58  CALL h5pset_layout_f(dcpl, h5d_compact_f, hdferr)
59  !
60  ! Create the dataset. We will use all default properties for this
61  ! example.
62  !
63  CALL h5dcreate_f(file, dataset, h5t_std_i32le, space, dset, hdferr, dcpl)
64  !
65  ! Write the data to the dataset.
66  !
67  CALL h5dwrite_f(dset, h5t_native_integer, wdata, dims, hdferr)
68  !
69  ! Close and release resources.
70  !
71  CALL h5pclose_f(dcpl , hdferr)
72  CALL h5dclose_f(dset , hdferr)
73  CALL h5sclose_f(space, hdferr)
74  CALL h5fclose_f(file , hdferr)
75  !
76  ! Now we begin the READ section of this example.
77  !
78  !
79  ! Open file and dataset using the default properties.
80  !
81  CALL h5fopen_f(filename, h5f_acc_rdonly_f, file, hdferr)
82  CALL h5dopen_f (file, dataset, dset, hdferr)
83  !
84  ! Retrieve the dataset creation property list, and print the
85  ! storage layout.
86  !
87  CALL h5dget_create_plist_f(dset, dcpl, hdferr)
88  CALL h5pget_layout_f(dcpl, layout, hdferr)
89  WRITE(*,'(/,"Storage layout for ", A," is: ")', advance='NO') dataset
90  IF(layout.EQ.h5d_compact_f)THEN
91  WRITE(*,'("H5D_COMPACT_F",/)')
92  ELSE IF (layout.EQ.h5d_contiguous_f)THEN
93  WRITE(*,'("H5D_CONTIGUOUS_F",/)')
94  ELSE IF (layout.EQ.h5d_chunked_f)THEN
95  WRITE(*,'("H5D_CHUNKED_F",/)')
96  ELSE IF (layout.EQ.h5d_virtual_f)THEN
97  WRITE(*,'("H5D_VIRTUAL_F",/)')
98  ELSE
99  WRITE(*,'("Layout Error",/)')
100  ENDIF
101  !
102  ! Read the data using the default properties.
103  !
104  CALL h5dread_f(dset, h5t_native_integer, rdata, dims, hdferr)
105  !
106  ! Output the data to the screen.
107  !
108  WRITE(*, '(A,":")') dataset
109  DO i=1, dim0
110  WRITE(*,'(" [")', advance='NO')
111  WRITE(*,'(80i3)', advance='NO') rdata(i,:)
112  WRITE(*,'(" ]")')
113  ENDDO
114  !
115  ! Close and release resources.
116  !
117  CALL h5pclose_f(dcpl , hdferr)
118  CALL h5dclose_f(dset , hdferr)
119  CALL h5fclose_f(file , hdferr)
120 
121 END PROGRAM main
