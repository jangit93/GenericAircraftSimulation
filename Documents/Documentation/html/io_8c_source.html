96 static void
97 mat_logfunc( int log_level, char *message )
98 {
99  if ( progname ) {
100  if ( log_level & MATIO_LOG_LEVEL_CRITICAL) {
101  fprintf(stderr,"-E- %s: %s\n", progname, message);
102  fflush(stderr);
103  } else if ( log_level & MATIO_LOG_LEVEL_ERROR ) {
104  fprintf(stderr,"-E- %s: %s\n", progname, message);
105  fflush(stderr);
106  abort();
107  } else if ( log_level & MATIO_LOG_LEVEL_WARNING ) {
108  fprintf(stderr,"-W- %s: %s\n", progname, message);
109  fflush(stderr);
110  } else if ( log_level & MATIO_LOG_LEVEL_DEBUG ) {
111  fprintf(stderr,"-D- %s: %s\n", progname, message);
112  fflush(stderr);
113  } else if ( log_level & MATIO_LOG_LEVEL_MESSAGE ) {
114  fprintf(stdout,"%s\n", message);
115  fflush(stdout);
116  }
117  } else {
118  if ( log_level & MATIO_LOG_LEVEL_CRITICAL) {
119  fprintf(stderr,"-E- : %s\n", message);
120  fflush(stderr);
121  } else if ( log_level & MATIO_LOG_LEVEL_ERROR ) {
122  fprintf(stderr,"-E- : %s\n", message);
123  fflush(stderr);
124  abort();
125  } else if ( log_level & MATIO_LOG_LEVEL_WARNING ) {
126  fprintf(stderr,"-W- : %s\n", message);
127  fflush(stderr);
128  } else if ( log_level & MATIO_LOG_LEVEL_DEBUG ) {
129  fprintf(stderr,"-D- : %s\n", message);
130  fflush(stderr);
131  } else if ( log_level & MATIO_LOG_LEVEL_MESSAGE ) {
132  fprintf(stdout,"%s\n", message);
133  fflush(stdout);
134  }
135  }
