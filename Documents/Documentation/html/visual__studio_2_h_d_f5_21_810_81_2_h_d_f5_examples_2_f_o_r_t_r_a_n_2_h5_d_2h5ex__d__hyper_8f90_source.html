1 !************************************************************
2 !
3 ! This example shows how to read and write data to a
4 ! dataset by hyberslabs. The program first writes integers
5 ! in a hyperslab selection to a dataset with dataspace
6 ! dimensions of DIM0xDIM1, then closes the file. Next, it
7 ! reopens the file, reads back the data, and outputs it to
8 ! the screen. Finally it reads the data again using a
9 ! different hyperslab selection, and outputs the result to
10 ! the screen.
11 !
12 ! This file is intended for use with HDF5 Library verion 1.8
13 !
14 ! ************************************************************/
15 PROGRAM main
16 
17  USE hdf5
18 
19  IMPLICIT NONE
20 
21  CHARACTER(LEN=15), PARAMETER :: filename = "h5ex_d_hyper.h5"
22  CHARACTER(LEN=3) , PARAMETER :: dataset = "DS1"
23  INTEGER , PARAMETER :: dim0 = 6
24  INTEGER , PARAMETER :: dim1 = 8
25 
26  INTEGER(HID_T) :: file, space, dset ! Handles
27  INTEGER :: hdferr
28  INTEGER(HSIZE_T), DIMENSION(1:2) :: dims = (/dim0, dim1/)
29  INTEGER(HSIZE_T), DIMENSION(1:2) :: start, stride, count, block
30 
31  INTEGER, DIMENSION(1:dim0, 1:dim1) :: wdata, & ! Write buffer
32  rdata ! Read buffer
33  INTEGER :: i, j
34 
35  !
36  ! Initialize FORTRAN interface.
37  !
38  CALL h5open_f(hdferr)
39  ! Initialize data to "1", to make it easier to see the selections.
40  !
41  wdata = 1
42  !
43  ! Print the data to the screen.
44  !
45  WRITE(*, '(/,"Original Data:")')
46  DO i=1, dim0
47  WRITE(*,'(" [")', advance='NO')
48  WRITE(*,'(80i3)', advance='NO') wdata(i,:)
49  WRITE(*,'(" ]")')
50  ENDDO
51  !
52  ! Create a new file using the default properties.
53  !
54  CALL h5fcreate_f(filename, h5f_acc_trunc_f, file, hdferr)
55  !
56  ! Create dataspace. Setting maximum size to be the current size.
57  !
58  CALL h5screate_simple_f(2, dims, space, hdferr)
59  ! Create the dataset. We will use all default properties for this
60  ! example.
61  !
62  CALL h5dcreate_f(file, dataset, h5t_std_i32le, space, dset, hdferr)
63  !
64  ! Define and select the first part of the hyperslab selection.
65  !
66  start = 0
67  stride = 3
68  count(1:2) = (/2,3/)
69  block = 2
70  CALL h5sselect_hyperslab_f (space, h5s_select_set_f, start, count, &
71  hdferr, stride, block)
72  !
73  ! Define and select the second part of the hyperslab selection,
74  ! which is subtracted from the first selection by the use of
75  ! H5S_SELECT_NOTB
76  !
77  block = 1
78  CALL h5sselect_hyperslab_f (space, h5s_select_notb_f, start, count, &
79  hdferr, stride, block)
80  !
81  ! Write the data to the dataset.
82  !
83  CALL h5dwrite_f(dset, h5t_native_integer, wdata, dims, hdferr, file_space_id=space)
84  !
85  ! Close and release resources.
86  !
87  CALL h5dclose_f(dset , hdferr)
88  CALL h5sclose_f(space, hdferr)
89  CALL h5fclose_f(file , hdferr)
90  !
91  ! Now we begin the read section of this example.
92  !
93  !
94  ! Open file and dataset using the default properties.
95  !
96  CALL h5fopen_f(filename, h5f_acc_rdonly_f, file, hdferr)
97  CALL h5dopen_f (file, dataset, dset, hdferr)
98  !
99  ! Read the data using the default properties.
100  !
101  CALL h5dread_f(dset, h5t_native_integer, rdata, dims, hdferr)
102  !
103  ! Output the data to the screen.
104  !
105  WRITE(*, '(/,"Data as written to disk by hyberslabs:")')
106  DO i=1, dim0
107  WRITE(*,'(" [")', advance='NO')
108  WRITE(*,'(80i3)', advance='NO') rdata(i,:)
109  WRITE(*,'(" ]")')
110  ENDDO
111  !
112  ! Initialize the read array.
113  !
114  rdata = 0
115  !
116  ! Define and select the hyperslab to use for reading.
117  !
118  CALL h5dget_space_f(dset, space, hdferr)
119 
120  start(1:2)=(/0,1/)
121  stride = 4
122  count = 2
123  block(1:2)=(/2,3/)
124 
125  CALL h5sselect_hyperslab_f (space, h5s_select_set_f, start, count, &
126  hdferr, stride, block)
127  !
128  ! Read the data using the previously defined hyperslab.
129  !
130  CALL h5dread_f(dset, h5t_native_integer, rdata, dims, hdferr, file_space_id=space)
131  !
132  ! Output the DATA to the screen.
133  !
134  WRITE(*, '(/,"Data as read from disk by hyberslabs:")')
135  DO i=1, dim0
136  WRITE(*,'(" [")', advance='NO')
137  WRITE(*,'(80i3)', advance='NO') rdata(i,:)
138  WRITE(*,'(" ]")')
139  ENDDO
140  WRITE(*,'(/)')
141  !
142  ! Close and release resources.
143  !
144  CALL h5dclose_f(dset , hdferr)
145  CALL h5sclose_f(space, hdferr)
146  CALL h5fclose_f(file , hdferr)
147 
148 END PROGRAM main
