1 /************************************************************
2 
3  This example shows how to create and extend an unlimited
4  dataset. The program first writes integers to a dataset
5  with dataspace dimensions of DIM0xDIM1, then closes the
6  file. Next, it reopens the file, reads back the data,
7  outputs it to the screen, extends the dataset, and writes
8  new data to the entire extended dataset. Finally it
9  reopens the file again, reads back the data, and utputs it
10  to the screen.
11 
12  This file is intended for use with HDF5 Library version 1.8
13 
14  ************************************************************/
15 
16 #include "hdf5.h"
17 #include <stdio.h>
18 #include <stdlib.h>
19 
20 #define FILE "h5ex_d_unlimmod.h5"
21 #define DATASET "DS1"
22 #define DIM0 4
23 #define DIM1 7
24 #define EDIM0 6
25 #define EDIM1 10
26 #define CHUNK0 4
27 #define CHUNK1 4
28 
29 int
30 main (void)
31 {
32  hid_t file, space, dset, dcpl; /* Handles */
33  herr_t status;
34  hsize_t dims[2] = {DIM0, DIM1},
35  extdims[2] = {EDIM0, EDIM1},
36  maxdims[2],
37  chunk[2] = {CHUNK0, CHUNK1};
38  int wdata[DIM0][DIM1], /* Write buffer */
39  wdata2[EDIM0][EDIM1], /* Write buffer for
40  extension */
41  **rdata, /* Read buffer */
42  ndims,
43  i, j;
44 
45  /*
46  * Initialize data.
47  */
48  for (i=0; i<DIM0; i++)
49  for (j=0; j<DIM1; j++)
50  wdata[i][j] = i * j - j;
51 
52  /*
53  * Create a new file using the default properties.
54  */
55  file = H5Fcreate (FILE, H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
56 
57  /*
58  * Create dataspace with unlimited dimensions.
59  */
60  maxdims[0] = H5S_UNLIMITED;
61  maxdims[1] = H5S_UNLIMITED;
62  space = H5Screate_simple (2, dims, maxdims);
63 
64  /*
65  * Create the dataset creation property list, and set the chunk
66  * size.
67  */
68  dcpl = H5Pcreate (H5P_DATASET_CREATE);
69  status = H5Pset_chunk (dcpl, 2, chunk);
70 
71  /*
72  * Create the unlimited dataset.
73  */
74  dset = H5Dcreate (file, DATASET, H5T_STD_I32LE, space, H5P_DEFAULT, dcpl,
75  H5P_DEFAULT);
76 
77  /*
78  * Write the data to the dataset.
79  */
80  status = H5Dwrite (dset, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT,
81  wdata[0]);
82 
83  /*
84  * Close and release resources.
85  */
86  status = H5Pclose (dcpl);
87  status = H5Dclose (dset);
88  status = H5Sclose (space);
89  status = H5Fclose (file);
90 
91 
92  /*
93  * In this next section we read back the data, extend the dataset,
94  * and write new data to the entire dataset.
95  */
96 
97  /*
98  * Open file and dataset using the default properties.
99  */
100  file = H5Fopen (FILE, H5F_ACC_RDWR, H5P_DEFAULT);
101  dset = H5Dopen (file, DATASET, H5P_DEFAULT);
102 
103  /*
104  * Get dataspace and allocate memory for read buffer. This is a
105  * two dimensional dataset so the dynamic allocation must be done
106  * in steps.
107  */
108  space = H5Dget_space (dset);
109  ndims = H5Sget_simple_extent_dims (space, dims, NULL);
110 
111  /*
112  * Allocate array of pointers to rows.
113  */
114  rdata = (int **) malloc (dims[0] * sizeof (int *));
115 
116  /*
117  * Allocate space for integer data.
118  */
119  rdata[0] = (int *) malloc (dims[0] * dims[1] * sizeof (int));
120 
121  /*
122  * Set the rest of the pointers to rows to the correct addresses.
123  */
124  for (i=1; i<dims[0]; i++)
125  rdata[i] = rdata[0] + i * dims[1];
126 
127  /*
128  * Read the data using the default properties.
129  */
130  status = H5Dread (dset, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT,
131  rdata[0]);
132 
133  /*
134  * Output the data to the screen.
135  */
136  printf ("Dataset before extension:\n");
137  for (i=0; i<dims[0]; i++) {
138  printf (" [");
139  for (j=0; j<dims[1]; j++)
140  printf (" %3d", rdata[i][j]);
141  printf ("]\n");
142  }
143 
144  /*
145  * Extend the dataset.
146  */
147  status = H5Dset_extent (dset, extdims);
148 
149  /*
150  * Initialize data for writing to the extended dataset.
151  */
152  for (i=0; i<EDIM0; i++)
153  for (j=0; j<EDIM1; j++)
154  wdata2[i][j] = j;
155 
156  /*
157  * Write the data to the extended dataset.
158  */
159  status = H5Dwrite (dset, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT,
160  wdata2[0]);
161 
162  /*
163  * Close and release resources.
164  */
165  free (rdata[0]);
166  free(rdata);
167  status = H5Dclose (dset);
168  status = H5Sclose (space);
169  status = H5Fclose (file);
170 
171 
172  /*
173  * Now we simply read back the data and output it to the screen.
174  */
175 
176  /*
177  * Open file and dataset using the default properties.
178  */
179  file = H5Fopen (FILE, H5F_ACC_RDONLY, H5P_DEFAULT);
180  dset = H5Dopen (file, DATASET, H5P_DEFAULT);
181 
182  /*
183  * Get dataspace and allocate memory for the read buffer as before.
184  */
185  space = H5Dget_space (dset);
186  ndims = H5Sget_simple_extent_dims (space, dims, NULL);
187  rdata = (int **) malloc (dims[0] * sizeof (int *));
188  rdata[0] = (int *) malloc (dims[0] * dims[1] * sizeof (int));
189  for (i=1; i<dims[0]; i++)
190  rdata[i] = rdata[0] + i * dims[1];
191 
192  /*
193  * Read the data using the default properties.
194  */
195  status = H5Dread (dset, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT,
196  rdata[0]);
197 
198  /*
199  * Output the data to the screen.
200  */
201  printf ("\nDataset after extension:\n");
202  for (i=0; i<dims[0]; i++) {
203  printf (" [");
204  for (j=0; j<dims[1]; j++)
205  printf (" %3d", rdata[i][j]);
206  printf ("]\n");
207  }
208 
209  /*
210  * Close and release resources.
211  */
212  free (rdata[0]);
213  free(rdata);
214  status = H5Dclose (dset);
215  status = H5Sclose (space);
216  status = H5Fclose (file);
217 
218  return 0;
219 }
