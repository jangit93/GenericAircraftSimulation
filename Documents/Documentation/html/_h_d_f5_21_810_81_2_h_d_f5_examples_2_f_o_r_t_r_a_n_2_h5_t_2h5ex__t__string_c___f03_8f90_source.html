1 !************************************************************
2 !
3 ! This example shows how to write a C string to a dataset
4 ! and read it back as a Fortran string.
5 ! The program first writes the C
6 ! strings to a dataset with a dataspace of DIM0, then closes the file.
7 ! Next, it reopens the file, reads back the data into a Fortran
8 ! fixed character string, and outputs it to the screen.
9 !
10 ! This file is intended for use with HDF5 Library version 1.8
11 ! with --enable-fortran2003
12 !
13 ! ************************************************************/
14 PROGRAM main
15 
16  USE hdf5
17  USE iso_c_binding
18 
19  IMPLICIT NONE
20 
21  CHARACTER(LEN=21), PARAMETER :: filename = "h5ex_t_stringC_F03.h5"
22  CHARACTER(LEN=3) , PARAMETER :: dataset = "DS1"
23  INTEGER , PARAMETER :: dim0 = 4
24  INTEGER(SIZE_T) , PARAMETER :: sdim = 7
25 
26  INTEGER(HID_T) :: file, filetype, memtype, space, dset ! Handles
27  INTEGER :: hdferr
28 
29  INTEGER(HSIZE_T), DIMENSION(1:1) :: dims = (/dim0/)
30  INTEGER(HSIZE_T), DIMENSION(1:1) :: maxdims
31 
32  CHARACTER(LEN=sdim), DIMENSION(1:dim0), TARGET :: &
33  wdata = (/"Parting", "is such", "sweet ", "sorrow."/)
34  CHARACTER(LEN=sdim), DIMENSION(:), ALLOCATABLE, TARGET :: rdata
35  INTEGER :: i
36  INTEGER(SIZE_T) :: size
37  TYPE(c_ptr) :: f_ptr
38  !
39  ! Initialize FORTRAN interface.
40  !
41  CALL h5open_f(hdferr)
42  !
43  ! Create a new file using the default properties.
44  !
45  CALL h5fcreate_f(filename, h5f_acc_trunc_f, file, hdferr)
46 
47  !
48  ! Create file and memory datatypes. For this example we will save
49  ! the strings as C strings
50  !
51  ! Include the NULL TERMINATION of string in C (i.e. add +1 to the length)
52 
53  CALL h5tcopy_f(h5t_c_s1, filetype, hdferr)
54  CALL h5tset_size_f(filetype, sdim+1, hdferr)
55 
56  CALL h5tcopy_f( h5t_fortran_s1, memtype, hdferr)
57  CALL h5tset_size_f(memtype, sdim, hdferr)
58  !
59  ! Create dataspace.
60  !
61  CALL h5screate_simple_f(1, dims, space, hdferr)
62  !
63  ! Create the dataset and write the string data to it.
64  !
65  CALL h5dcreate_f(file, dataset, filetype, space, dset, hdferr)
66 
67  f_ptr = c_loc(wdata(1)(1:1))
68  CALL h5dwrite_f(dset, memtype, f_ptr, hdferr)
69 
70  !
71  ! Close and release resources.
72  !
73  CALL h5dclose_f(dset , hdferr)
74  CALL h5sclose_f(space, hdferr)
75  CALL h5tclose_f(filetype, hdferr)
76  CALL h5tclose_f(memtype, hdferr)
77  CALL h5fclose_f(file , hdferr)
78  !
79  ! Now we begin the read section of this example.
80  !
81  ! Open file and dataset.
82  !
83  CALL h5fopen_f(filename, h5f_acc_rdonly_f, file, hdferr)
84  CALL h5dopen_f(file, dataset, dset, hdferr)
85  !
86  ! Get the datatype and its size.
87  !
88  CALL h5dget_type_f(dset, filetype, hdferr)
89  CALL h5tget_size_f(filetype, size, hdferr)
90 
91  ! Make sure the declared length is large enough,
92  ! the C string contains the null character.
93  IF(size.GT.sdim+1)THEN
94  print*,'ERROR: Character LEN is to small'
95  stop
96  ENDIF
97 
98  ! Get dataspace.
99  !
100  CALL h5dget_space_f(dset, space, hdferr)
101  CALL h5sget_simple_extent_dims_f(space, dims, maxdims, hdferr)
102 
103  ALLOCATE(rdata(1:dims(1)))
104  !
105  ! Create the memory datatype.
106  !
107  CALL h5tcopy_f(h5t_fortran_s1, memtype, hdferr)
108  CALL h5tset_size_f(memtype, sdim, hdferr)
109  !
110  ! Read the data.
111  !
112  f_ptr = c_loc(rdata(1)(1:1))
113  CALL h5dread_f(dset, memtype, f_ptr, hdferr, space)
114  !
115  ! Output the data to the screen.
116  !
117  DO i = 1, dims(1)
118  WRITE(*,'(A,"(",I0,"): ", A)') dataset, i, rdata(i)
119  END DO
120  !
121  ! Close and release resources.
122  !
123  CALL h5dclose_f(dset, hdferr)
124  CALL h5sclose_f(space, hdferr)
125  CALL h5tclose_f(filetype, hdferr)
126  CALL h5tclose_f(memtype, hdferr)
127  CALL h5fclose_f(file, hdferr)
128 
129  DEALLOCATE(rdata)
130 
131 
132 END PROGRAM main
