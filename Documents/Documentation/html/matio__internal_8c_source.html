125 {
126  char *fname;

128 
129  fname = fstrdup(filename,len);
130 
131  if ( NULL == (mat->mat_t_c_ptr = Mat_Open(fname,*mode)) ) {
132  Mat_Critical("Error opening file %s", fname);

136  /* V-4 matlab files don't have a header */

141  }
142  }
143  free(fname);
144 
145  return err;

276 fmat_varreaddata_c(struct fmat_t *mat,struct fmatvar_t *matvar,char *data,int *start,int *stride,int *edge)
277 {
278  int *start_c, *stride_c, *edge_c, err, i;
279 
280  if ( (start == NULL) || (stride == NULL) || (edge == NULL) ) {
281  start_c = malloc(matvar->rank*sizeof(int));
282  stride_c = malloc(matvar->rank*sizeof(int));
283  edge_c = malloc(matvar->rank*sizeof(int));
284  if ( (start_c == NULL) || (stride_c == NULL) || (edge_c == NULL) ) {
285  err = 1;
286  } else {
287  for ( i = 0; i < matvar->rank; i++ ) {
288  start_c[i] = 0;
289  stride_c[i] = 1;
290  edge_c[i] = matvar->dims[i];
291  }
292  err = Mat_VarReadData(mat->mat_t_c_ptr,matvar->matvar_t_c_ptr,(data),start_c,
293  stride_c,edge_c);
294  free(start);
295  free(stride);
296  free(edge);
297  }
298  } else {
299  err = Mat_VarReadData(mat->mat_t_c_ptr,matvar->matvar_t_c_ptr,(data),start,stride,
300  edge);
301  }
302 
303  return err;
304 }
