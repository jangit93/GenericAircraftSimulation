39 {
40  cs* A = cs_transpose (a, 1) ;
41  cs* B = cs_transpose (b, 1) ;
42  cs* D = cs_multiply (B,A) ; /* D = B'*A' */
43  cs_spfree (A) ;
44  cs_spfree (B) ;
45  cs_dropzeros (D) ; /* drop zeros from D */
46  cs* C = cs_transpose (D, 1) ; /* C = D', so that C is sorted */
