252 /*
253  * "Generic" chunk record. Each chunk is keyed by the minimum logical
254  * N-dimensional coordinates and the datatype size of the chunk.
255  * The fastest-varying dimension is assumed to reference individual bytes of
256  * the array, so a 100-element 1-D array of 4-byte integers would really be a
257  * 2-D array with the slow varying dimension of size 100 and the fast varying
258  * dimension of size 4 (the storage dimensionality has very little to do with
259  * the real dimensionality).
260  *
261  * The chunk's file address, filter mask and size on disk are not key values.
262  */
263 typedef struct H5D_chunk_rec_t {
264  hsize_t scaled[H5O_LAYOUT_NDIMS]; /* Logical offset to start */
265  uint32_t nbytes; /* Size of stored data */
266  uint32_t filter_mask; /* Excluded filters */
267  haddr_t chunk_addr; /* Address of chunk in file */
268 } H5D_chunk_rec_t;
