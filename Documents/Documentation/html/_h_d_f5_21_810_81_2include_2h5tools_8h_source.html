209  * Information about how to format output.

211 typedef struct h5tool_format_t {
212  /*
213  * Fields associated with formatting numeric data. If a datatype matches
214  * multiple formats based on its size, then the first applicable format
215  * from this list is used. However, if `raw' is non-zero then dump all
216  * data in hexadecimal format without translating from what appears on
217  * disk.
218  *
219  * raw: If set then print all data as hexadecimal without
220  * performing any conversion from disk.
221  *
222  * fmt_raw: The printf() format for each byte of raw data. The
223  * default is `%02x'.
224  *
225  * fmt_int: The printf() format to use when rendering data which is
226  * typed `int'. The default is `%d'.
227  *
228  * fmt_uint: The printf() format to use when rendering data which is
229  * typed `unsigned'. The default is `%u'.
230  *
231  * fmt_schar: The printf() format to use when rendering data which is
232  * typed `signed char'. The default is `%d'. This format is
233  * used ony if the `ascii' field is zero.
234  *
235  * fmt_uchar: The printf() format to use when rendering data which is
236  * typed `unsigned char'. The default is `%u'. This format
237  * is used only if the `ascii' field is zero.
238  *
239  * fmt_short: The printf() format to use when rendering data which is
240  * typed `short'. The default is `%d'.
241  *
242  * fmt_ushort: The printf() format to use when rendering data which is
243  * typed `unsigned short'. The default is `%u'.
244  *
245  * fmt_long: The printf() format to use when rendering data which is
246  * typed `long'. The default is `%ld'.
247  *
248  * fmt_ulong: The printf() format to use when rendering data which is
249  * typed `unsigned long'. The default is `%lu'.
250  *
251  * fmt_llong: The printf() format to use when rendering data which is
252  * typed `long long'. The default depends on what printf()
253  * format is available to print this datatype.
254  *
255  * fmt_ullong: The printf() format to use when rendering data which is
256  * typed `unsigned long long'. The default depends on what
257  * printf() format is available to print this datatype.
258  *
259  * fmt_double: The printf() format to use when rendering data which is
260  * typed `double'. The default is `%g'.
261  *
262  * fmt_float: The printf() format to use when rendering data which is
263  * typed `float'. The default is `%g'.
264  *
265  * ascii: If set then print 1-byte integer values as an ASCII
266  * character (no quotes). If the character is one of the
267  * standard C escapes then print the escaped version. If
268  * the character is unprintable then print a 3-digit octal
269  * escape. If `ascii' is zero then then 1-byte integers are
270  * printed as numeric values. The default is zero.
271  *
272  * str_locale: Determines how strings are printed. If zero then strings
273  * are printed like in C except. If set to ESCAPE_HTML then
274  * strings are printed using HTML encoding where each
275  * character not in the class [a-zA-Z0-9] is substituted
276  * with `%XX' where `X' is a hexadecimal digit.
277  *
278  * str_repeat: If set to non-zero then any character value repeated N
279  * or more times is printed as 'C'*N
280  *
281  * Numeric data is also subject to the formats for individual elements.
282  */
283  hbool_t raw;
284  const char *fmt_raw;
285  const char *fmt_int;
286  const char *fmt_uint;
287  const char *fmt_schar;
288  const char *fmt_uchar;
289  const char *fmt_short;
290  const char *fmt_ushort;
291  const char *fmt_long;
292  const char *fmt_ulong;
293  const char *fmt_llong;
294  const char *fmt_ullong;
295  const char *fmt_double;
296  const char *fmt_float;
297  int ascii;
298  int str_locale;
299  unsigned str_repeat;

301  /*
302  * Fields associated with compound array members.
303  *
304  * pre: A string to print at the beginning of each array. The
305  * default value is the left square bracket `['.
306  *
307  * sep: A string to print between array values. The default
308  * value is a ",\001" ("\001" indicates an optional line
309  * break).
310  *
311  * suf: A string to print at the end of each array. The default
312  * value is a right square bracket `]'.
313  *
314  * linebreaks: a boolean value to determine if we want to break the line
315  * after each row of an array.
316  */
317  const char *arr_pre;
318  const char *arr_sep;
319  const char *arr_suf;

322  /*
323  * Fields associated with compound data types.
324  *
325  * name: How the name of the struct member is printed in the
326  * values. By default the name is not printed, but a
327  * reasonable setting might be "%s=" which prints the name
328  * followed by an equal sign and then the value.
329  *
330  * sep: A string that separates one member from another. The
331  * default is ", \001" (the \001 indicates an optional
332  * line break to allow structs to span multiple lines of
333  * output).
334  *
335  * pre: A string to print at the beginning of a compound type.
336  * The default is a left curly brace.
337  *
338  * suf: A string to print at the end of each compound type. The
339  * default is right curly brace.
340  *
341  * end: a string to print after we reach the last element of
342  * each compound type. prints out before the suf.
343  *
344  * listv: h5watch: vector containing info about the list of compound fields to be printed.
345  */

361  *

381  *
382  * suf1: This string is appended to elements which are followed by
383  * another element whether the following element is on the
384  * same line or the next line. The default is a comma.
385  *

394  /*
395  * Fields associated with the index values printed at the left edge of
396  * each line of output.
397  *
398  * n_fmt: Each index value is printed according to this printf(3c)
399  * format string which should include a format for a long
400  * integer. The default is "%lu".
401  *
402  * sep: Each integer in the index list will be separated from the
403  * others by this string, which defaults to a comma.
404  *
405  * fmt: After the index values are formated individually and
406  * separated from one another by some string, the entire
407  * resulting string will be formated according to this
408  * printf(3c) format which should include a format for a
409  * character string. The default is "%s".
410  */
411  const char *idx_n_fmt; /*index number format */
412  const char *idx_sep; /*separator between numbers */
413  const char *idx_fmt; /*entire index format */

415  /*
416  * Fields associated with entire lines.
417  *
418  * ncols: Number of columns per line defaults to 80.
419  *
420  * per_line: If this field has a positive value then every Nth element
421  * will be printed at the beginning of a line.
422  *
423  * pre: Each line of output contains an optional prefix area
424  * before the data. This area can contain the index for the
425  * first datum (represented by `%s') as well as other
426  * constant text. The default value is `%s'.
427  *
428  * 1st: This is the format to print at the beginning of the first
429  * line of output. The default value is the current value of
430  * `pre' described above.
431  *
432  * cont: This is the format to print at the beginning of each line
433  * which was continued because the line was split onto
434  * multiple lines. This often happens with compound
435  * data which is longer than one line of output. The default
436  * value is the current value of the `pre' field
437  * described above.
438  *
439  * suf: This character string will be appended to each line of
440  * output. It should not contain line feeds. The default
441  * is the empty string.
442  *
443  * sep: A character string to be printed after every line feed
444  * defaulting to the empty string. It should end with a
445  * line feed.
446  *
447  * multi_new: Indicates the algorithm to use when data elements tend to
448  * occupy more than one line of output. The possible values
449  * are (zero is the default):
450  *
451  * 0: No consideration. Each new element is printed
452  * beginning where the previous element ended.
453  *
454  * 1: Print the current element beginning where the
455  * previous element left off. But if that would result
456  * in the element occupying more than one line and it
457  * would only occupy one line if it started at the
458  * beginning of a line, then it is printed at the
459  * beginning of the next line.
460  *
461  * multi_new: If an element is continued onto additional lines then
462  * should the following element begin on the next line? The
463  * default is to start the next element on the same line
464  * unless it wouldn't fit.
465  *
466  * indentlevel: a string that shows how far to indent if extra spacing
467  * is needed. dumper uses it.
468  */
469  unsigned line_ncols; /*columns of output */
470  size_t line_per_line; /*max elements per line */
471  const char *line_pre; /*prefix at front of each line */
472  const char *line_1st; /*alternate pre. on first line */

474  const char *line_suf; /*string to append to each line */
