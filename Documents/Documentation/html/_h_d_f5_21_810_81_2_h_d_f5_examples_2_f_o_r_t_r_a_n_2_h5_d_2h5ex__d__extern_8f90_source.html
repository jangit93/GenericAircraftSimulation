1 !************************************************************
2 !
3 ! This example shows how to read and write data to an
4 ! external dataset. The program first writes integers to an
5 ! external dataset with dataspace dimensions of DIM0xDIM1,
6 ! then closes the file. Next, it reopens the file, reads
7 ! back the data, and outputs the name of the external data
8 ! file and the data to the screen.
9 !
10 ! This file is intended for use with HDF5 Library verion 1.8
11 !
12 ! ************************************************************/
13 
14 PROGRAM main
15 
16  USE hdf5
17 
18  IMPLICIT NONE
19 
20  CHARACTER(LEN=16), PARAMETER :: filename = "h5ex_d_extern.h5"
21  CHARACTER(LEN=18), PARAMETER :: externalname = "h5ex_d_extern.data"
22  CHARACTER(LEN=3) , PARAMETER :: dataset = "DS1"
23  INTEGER , PARAMETER :: dim0 = 4
24  INTEGER , PARAMETER :: dim1 = 7
25  INTEGER(SIZE_T) , PARAMETER :: name_buf_size = 32
26 
27  INTEGER :: hdferr
28  INTEGER :: layout
29  INTEGER(HID_T) :: file, space, dset, dcpl ! Handles
30  INTEGER(HSIZE_T), DIMENSION(1:2) :: dims = (/dim0, dim1/)
31  INTEGER(HSIZE_T), DIMENSION(1:2) :: start, stride, count, block
32 
33  INTEGER, DIMENSION(1:dim0, 1:dim1) :: wdata, & ! Write buffer
34  rdata ! Read buffer
35  CHARACTER(LEN=name_buf_size) :: name
36  INTEGER :: i, j
37  INTEGER(OFF_T) :: offset ! Offset, in bytes, from thebeginning of the file to the
38  ! location in the file where the data starts.
39  INTEGER(HSIZE_T) :: bytes ! Number of bytes reserved in the file for the data
40  INTEGER(SIZE_T) :: int_size ! size of integer
41  !
42  ! Initialize FORTRAN interface.
43  !
44  CALL h5open_f(hdferr)
45  !
46  ! Initialize data.
47  !
48  DO i = 1, dim0
49  DO j = 1, dim1
50  wdata(i,j) = (i-1)*(j-1)-(j-1)
51  ENDDO
52  ENDDO
53  !
54  ! Create a new file using the default properties.
55  !
56  CALL h5fcreate_f(filename, h5f_acc_trunc_f, file, hdferr)
57  !
58  ! Create dataspace. Setting maximum size to NULL sets the maximum
59  ! size to be the current size.
60  !
61  CALL h5screate_simple_f(2, dims, space, hdferr)
62  !
63  ! Create the dataset creation property list, set the external
64  ! file.
65  !
66  CALL h5pcreate_f(h5p_dataset_create_f, dcpl, hdferr)
67  ! for HDF5 versions <= 1.8.2 use:
68  CALL h5tget_size_f(h5t_native_integer, int_size, hdferr)
69  bytes = int_size*dim0*dim1
70  ! else use:
71  ! bytes = INT(H5F_UNLIMITED_F,HSIZE_T)
72  CALL h5pset_external_f(dcpl, externalname, int(0,off_t), bytes, hdferr)
73  !
74  ! Create the external dataset.
75  !
76  CALL h5dcreate_f(file, dataset, h5t_std_i32le, space, dset, hdferr, dcpl)
77  !
78  ! Write the data to the dataset.
79  !
80  CALL h5dwrite_f(dset, h5t_native_integer, wdata, dims, hdferr)
81  !
82  ! Close and release resources.
83  !
84  CALL h5pclose_f(dcpl , hdferr)
85  CALL h5dclose_f(dset , hdferr)
86  CALL h5sclose_f(space, hdferr)
87  CALL h5fclose_f(file , hdferr)
88  !
89  ! Now we begin the read section of this example.
90  !
91  !
92  ! Open file and dataset using the default properties.
93  !
94  CALL h5fopen_f(filename, h5f_acc_rdonly_f, file, hdferr)
95  CALL h5dopen_f(file, dataset, dset, hdferr)
96  !
97  ! Retrieve dataset creation property list.
98  !
99  CALL h5dget_create_plist_f(dset, dcpl, hdferr)
100  !
101  ! Retrieve and print the name of the external file. Here we
102  ! manually set the last field in name to null, in case the name of
103  ! the file is longer than the buffer.
104  !
105  CALL h5pget_external_f (dcpl, 0, name_buf_size, name, offset, bytes, hdferr)
106  WRITE(*,'(A," is stored in file: ",A)') dataset, trim(name)
107  !
108  ! Read the data using the default properties.
109  !
110  CALL h5dread_f(dset, h5t_native_integer, rdata, dims, hdferr)
111  !
112  ! Output the data to the screen.
113  !
114  WRITE(*, '(A,":")') dataset
115  DO i=1, dim0
116  WRITE(*,'(" [")', advance='NO')
117  WRITE(*,'(80i3)', advance='NO') rdata(i,:)
118  WRITE(*,'(" ]")')
119  ENDDO
120  !
121  ! Close and release resources.
122  !
123  CALL h5pclose_f(dcpl , hdferr)
124  CALL h5dclose_f(dset , hdferr)
125  CALL h5fclose_f(file , hdferr)
126 
127 END PROGRAM main
