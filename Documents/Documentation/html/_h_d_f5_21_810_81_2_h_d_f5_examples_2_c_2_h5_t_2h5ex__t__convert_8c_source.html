3  This example shows how to convert between different
4  datatypes in memory. The program converts DIM0 elements
5  of compound type sourcetype to desttype, then outputs the
6  converted data to the screen. A background buffer is used
7  to fill in the elements of desttype that are not in
8  sourcetype.
9 
10  This file is intended for use with HDF5 Library version 1.8
11 
12  ************************************************************/
13 
14 #include "hdf5.h"
15 #include <stdio.h>
16 #include <stdlib.h>
17 
18 #define DIM0 4
19 
20 typedef struct {
21  double temperature;
22  double pressure;
23 } reading_t; /* Source type */
24 
25 typedef struct {
26  int serial_no;
27  char *location;
28  double temperature;
29  double pressure;
30 } sensor_t; /* Destination type */
31 
32 int
33 main (void)
34 {
35  hid_t sourcetype, desttype, strtype, space;
36  /* Handles */
37  herr_t status;
38  hsize_t dims[1] = {DIM0};
39  reading_t *reading; /* Conversion buffer */
40  sensor_t *sensor, /* Conversion buffer */
41  bkgrd[DIM0]; /* Background buffer */
42  int i;
43 
44  /*
45  * Allocate memory for conversion buffer. We will allocate space
46  * for it to hold DIM0 elements of the destination type, as the
47  * type conversion is performed in place. Of course, if the
48  * destination type were smaller than the source type, we would
49  * allocate space to hold DIM0 elements of the source type.
50  */
51  reading = (reading_t *) malloc (DIM0 * sizeof (sensor_t));
52 
53  /*
54  * Assign the allocated space to a pointer of the destination type,
55  * to allow the buffer to be accessed correctly after the
56  * conversion has taken place.
57  */
58  sensor = (sensor_t *) reading;
59 
60  /*
61  * Initialize data.
62  */
63  bkgrd[0].serial_no = 1153;
64  bkgrd[0].location = "Exterior (static)";
65  bkgrd[0].temperature = 53.23;
66  bkgrd[0].pressure = 24.57;
67  bkgrd[1].serial_no = 1184;
68  bkgrd[1].location = "Intake";
69  bkgrd[1].temperature = 55.12;
70  bkgrd[1].pressure = 22.95;
71  bkgrd[2].serial_no = 1027;
72  bkgrd[2].location = "Intake manifold";
73  bkgrd[2].temperature = 103.55;
74  bkgrd[2].pressure = 31.23;
75  bkgrd[3].serial_no = 1313;
76  bkgrd[3].location = "Exhaust manifold";
77  bkgrd[3].temperature = 1252.89;
78  bkgrd[3].pressure = 84.11;
79 
80  reading[0].temperature = 54.84;
81  reading[0].pressure = 24.76;
82  reading[1].temperature = 56.63;
83  reading[1].pressure = 23.10;
84  reading[2].temperature = 102.69;
85  reading[2].pressure = 30.97;
86  reading[3].temperature = 1238.27;
87  reading[3].pressure = 82.15;
88 
89  /*
90  * Create variable-length string datatype.
91  */
92  strtype = H5Tcopy (H5T_C_S1);
93  status = H5Tset_size (strtype, H5T_VARIABLE);
94 
95  /*
96  * Create the compound datatype for memory.
97  */
98  sourcetype = H5Tcreate (H5T_COMPOUND, sizeof (reading_t));
99  status = H5Tinsert (sourcetype, "Temperature (F)",
100  HOFFSET (reading_t, temperature), H5T_NATIVE_DOUBLE);
101  status = H5Tinsert (sourcetype, "Pressure (inHg)",
102  HOFFSET (reading_t, pressure), H5T_NATIVE_DOUBLE);
103 
104  desttype = H5Tcreate (H5T_COMPOUND, sizeof (sensor_t));
105  status = H5Tinsert (desttype, "Serial number",
106  HOFFSET (sensor_t, serial_no), H5T_NATIVE_INT);
107  status = H5Tinsert (desttype, "Location", HOFFSET (sensor_t, location),
108  strtype);
109  status = H5Tinsert (desttype, "Temperature (F)",
110  HOFFSET (sensor_t, temperature), H5T_NATIVE_DOUBLE);
111  status = H5Tinsert (desttype, "Pressure (inHg)",
112  HOFFSET (sensor_t, pressure), H5T_NATIVE_DOUBLE);
113 
114  /*
115  * Create dataspace. Setting maximum size to NULL sets the maximum
116  * size to be the current size.
117  */
118  space = H5Screate_simple (1, dims, NULL);
119 
120  /*
121  * Convert the buffer in reading from sourcetype to desttype.
122  * After this conversion we will use sensor to access the buffer,
123  * as the buffer now matches its type.
124  */
125  status = H5Tconvert (sourcetype, desttype, DIM0, reading, bkgrd,
126  H5P_DEFAULT);
127 
128  /*
129  * Output the data to the screen.
130  */
131  for (i=0; i<DIM0; i++) {
132  printf ("sensor[%d]:\n", i);
133  printf ("Serial number : %d\n", sensor[i].serial_no);
134  printf ("Location : %s\n", sensor[i].location);
135  printf ("Temperature (F) : %f\n", sensor[i].temperature);
136  printf ("Pressure (inHg) : %f\n\n", sensor[i].pressure);
137  }
138 
139  /*
140  * Close and release resources. In this case H5Tconvert preserves
141  * the memory locations of the variable-length strings in
142  * "location", so we do not need to free those strings as they were
143  * initialized as string constants.
144  */
145  free (sensor);
146  status = H5Sclose (space);
147  status = H5Tclose (sourcetype);
148  status = H5Tclose (desttype);
149  status = H5Tclose (strtype);
150 
151  return 0;
152 }
