309 mat_t *
310 Mat_Open(const char *matname,int mode)
311 {
312  FILE *fp = NULL;
313  mat_int16_t tmp, tmp2;
314  mat_t *mat = NULL;
315  size_t bytesread = 0;
316 
317  if ( (mode & 0x01) == MAT_ACC_RDONLY ) {
318  fp = fopen( matname, "rb" );
319  if ( !fp )
320  return NULL;
321  } else if ( (mode & 0x01) == MAT_ACC_RDWR ) {
322  fp = fopen( matname, "r+b" );
323  if ( !fp ) {
324  mat = Mat_CreateVer(matname,NULL,(enum mat_ft)(mode&0xfffffffe));
325  return mat;
326  }
327  } else {
328  Mat_Critical("Invalid file open mode");
329  return NULL;
330  }
331 
332  mat = (mat_t*)malloc(sizeof(*mat));
333  if ( NULL == mat ) {
334  fclose(fp);
335  Mat_Critical("Couldn't allocate memory for the MAT file");
336  return NULL;
337  }
338 
339  mat->fp = fp;
340  mat->header = (char*)calloc(128,sizeof(char));
341  if ( NULL == mat->header ) {
342  free(mat);
343  fclose(fp);
344  Mat_Critical("Couldn't allocate memory for the MAT file header");
345  return NULL;
346  }
347  mat->subsys_offset = (char*)calloc(8,sizeof(char));
348  if ( NULL == mat->subsys_offset ) {
349  free(mat->header);
350  free(mat);
351  fclose(fp);
352  Mat_Critical("Couldn't allocate memory for the MAT file subsys offset");
353  return NULL;
354  }
355  mat->filename = NULL;
356  mat->version = 0;
357  mat->byteswap = 0;
358  mat->num_datasets = 0;
359 #if defined(MAT73) && MAT73
360  mat->refs_id = -1;
361 #endif
362  mat->dir = NULL;
363 
364  bytesread += fread(mat->header,1,116,fp);
365  mat->header[116] = '\0';
366  bytesread += fread(mat->subsys_offset,1,8,fp);
367  bytesread += 2*fread(&tmp2,2,1,fp);
368  bytesread += fread(&tmp,1,2,fp);
369 
370  if ( 128 == bytesread ) {
371  /* v5 and v7.3 files have at least 128 byte header */
372  mat->byteswap = -1;
373  if ( tmp == 0x4d49 )
374  mat->byteswap = 0;
375  else if ( tmp == 0x494d ) {
376  mat->byteswap = 1;
377  Mat_int16Swap(&tmp2);
378  }
379 
380  mat->version = (int)tmp2;
381  if ( (mat->version == 0x0100 || mat->version == 0x0200) &&
382  -1 != mat->byteswap ) {
383  mat->bof = ftell((FILE*)mat->fp);
384  if ( mat->bof == -1L ) {
385  free(mat->header);
386  free(mat->subsys_offset);
387  free(mat);
388  fclose(fp);
389  Mat_Critical("Couldn't determine file position");
390  return NULL;
391  }
392  mat->next_index = 0;
393  } else {
394  mat->version = 0;
395  }
396  }
397 
398  if ( 0 == mat->version ) {
399  /* Maybe a V4 MAT file */
400  matvar_t *var;
401 
402  free(mat->header);
403  free(mat->subsys_offset);
404 
405  mat->header = NULL;
406  mat->subsys_offset = NULL;
407  mat->fp = fp;
408  mat->version = MAT_FT_MAT4;
409  mat->byteswap = 0;
410  mat->mode = mode;
411  mat->bof = 0;
412  mat->next_index = 0;
413 #if defined(MAT73) && MAT73
414  mat->refs_id = -1;
415 #endif
416 
417  Mat_Rewind(mat);
418  var = Mat_VarReadNextInfo4(mat);
419  if ( NULL == var &&
420  bytesread != 0 ) { /* Accept 0 bytes files as a valid V4 file */
421  /* Does not seem to be a valid V4 file */
422  Mat_Close(mat);
423  mat = NULL;
424  Mat_Critical("\"%s\" does not seem to be a valid MAT file",matname);
425  } else {
426  Mat_VarFree(var);
427  Mat_Rewind(mat);
428  }
429  }
430 
431  if ( NULL == mat )
432  return mat;
433 
434  mat->filename = strdup_printf("%s",matname);
435  mat->mode = mode;
436 
437  if ( mat->version == 0x0200 ) {
438  fclose((FILE*)mat->fp);
439 #if defined(MAT73) && MAT73
440  mat->fp = malloc(sizeof(hid_t));
441 
442  if ( (mode & 0x01) == MAT_ACC_RDONLY )
443  *(hid_t*)mat->fp=H5Fopen(mat->filename,H5F_ACC_RDONLY,H5P_DEFAULT);
444  else if ( (mode & 0x01) == MAT_ACC_RDWR )
445  *(hid_t*)mat->fp=H5Fopen(mat->filename,H5F_ACC_RDWR,H5P_DEFAULT);
446 
447  if ( -1 < *(hid_t*)mat->fp ) {
448  H5G_info_t group_info = {0};
449  H5Gget_info(*(hid_t*)mat->fp, &group_info);
450  mat->num_datasets = (size_t)group_info.nlinks;
451  mat->refs_id = -1;
452  }
453 #else
454  mat->fp = NULL;
455  Mat_Close(mat);
456  mat = NULL;
457  Mat_Critical("No HDF5 support which is required to read the v7.3 "
458  "MAT file \"%s\"",matname);
459 #endif
460  }
461 
462  return mat;
463 }

551 Mat_GetDir(mat_t *mat, size_t *n)
552 {
553  char ** dir = NULL;
554 
555  if ( NULL == n )
556  return dir;
557 
558  if ( NULL == mat ) {
559  *n = 0;
560  return dir;
561  }
562 
563  if ( NULL == mat->dir ) {
564  matvar_t *matvar = NULL;
565 
566  if ( mat->version == MAT_FT_MAT73 ) {
567  size_t i = 0;
568  size_t fpos = mat->next_index;
569  if ( mat->num_datasets == 0 ) {
570  *n = 0;
571  return dir;
572  }
573  mat->dir = calloc(mat->num_datasets, sizeof(char*));
574  if ( NULL == mat->dir ) {
575  *n = 0;
576  Mat_Critical("Couldn't allocate memory for the directory");
577  return dir;
578  }
579  mat->next_index = 0;
580  while ( mat->next_index < mat->num_datasets ) {
581  matvar = Mat_VarReadNextInfo(mat);
582  if ( NULL != matvar ) {
583  if ( NULL != matvar->name ) {
584  mat->dir[i++] = strdup_printf("%s",
585  matvar->name);
586  }
587  Mat_VarFree(matvar);
588  } else {
589  Mat_Critical("An error occurred in reading the MAT file");
590  break;
591  }
592  }
593  mat->next_index = fpos;
594  *n = i;
595  } else {
596  long fpos = ftell((FILE*)mat->fp);
597  if ( fpos == -1L ) {
598  *n = 0;
599  Mat_Critical("Couldn't determine file position");
600  return dir;
601  }
602  (void)fseek((FILE*)mat->fp,mat->bof,SEEK_SET);
603  mat->num_datasets = 0;
604  do {
605  matvar = Mat_VarReadNextInfo(mat);
606  if ( NULL != matvar ) {
607  if ( NULL != matvar->name ) {
608  if ( NULL == mat->dir ) {
609  dir = malloc(sizeof(char*));
610  } else {
611  dir = realloc(mat->dir,
612  (mat->num_datasets + 1)*(sizeof(char*)));
613  }
614  if ( NULL != dir ) {
615  mat->dir = dir;
616  mat->dir[mat->num_datasets++] =
617  strdup_printf("%s", matvar->name);
618  } else {
619  Mat_Critical("Couldn't allocate memory for the directory");
620  break;
621  }
622  }
623  Mat_VarFree(matvar);
624  } else if ( !feof((FILE *)mat->fp) ) {
625  Mat_Critical("An error occurred in reading the MAT file");
626  break;
627  }
628  } while ( !feof((FILE *)mat->fp) );
629  (void)fseek((FILE*)mat->fp,fpos,SEEK_SET);
630  *n = mat->num_datasets;
631  }
632  }
633  dir = mat->dir;
634  return dir;
635 }
