3  This example shows a way to recursively traverse the file
4  using H5Literate. The method shown here guarantees that
5  the recursion will not enter an infinite loop, but does
6  not prevent objects from being visited more than once.
7  The program prints the directory structure of the file
8  specified in FILE. The default file used by this example
9  implements the structure described in the User's Guide,
10  chapter 4, figure 26.
11 
12  This file is intended for use with HDF5 Library version 1.8
13

21 /*
22  * Define operator data structure type for H5Literate callback.
23  * During recursive iteration, these structures will form a
24  * linked list that can be searched for duplicate groups,
25  * preventing infinite recursion.
26  */
27 struct opdata {
28  unsigned recurs; /* Recursion level. 0=root */
29  struct opdata *prev; /* Pointer to previous opdata */
30  haddr_t addr; /* Group address */
31 };

34  * Operator function to be called by H5Literate.
35  */

37  void *operator_data);
38 
39 /*
40  * Function to check for duplicate groups in a path.
41  */
42 int group_check (struct opdata *od, haddr_t target_addr);

78 /************************************************************
79 
80  Operator function. This function prints the name and type
81  of the object passed to it. If the object is a group, it
82  is first checked against other groups in its path using
83  the group_check function, then if it is not a duplicate,
84  H5Literate is called for that group. This guarantees that
85  the program will not enter infinite recursion due to a
86  circular path in the file.
87 
88  ************************************************************/
89 herr_t op_func (hid_t loc_id, const char *name, const H5L_info_t *info,
90  void *operator_data)
91 {
92  herr_t status, return_val = 0;
93  H5O_info_t infobuf;
94  struct opdata *od = (struct opdata *) operator_data;
95  /* Type conversion */
96  unsigned spaces = 2*(od->recurs+1);
97  /* Number of whitespaces to prepend
98  to output */
99 
100  /*
101  * Get type of the object and display its name and type.
102  * The name of the object is passed to this function by
103  * the Library.
104  */
105  status = H5Oget_info_by_name (loc_id, name, &infobuf, H5P_DEFAULT);
106  printf ("%*s", spaces, ""); /* Format output */
107  switch (infobuf.type) {
108  case H5O_TYPE_GROUP:
109  printf ("Group: %s {\n", name);
110 
111  /*
112  * Check group address against linked list of operator
113  * data structures. We will always run the check, as the
114  * reference count cannot be relied upon if there are
115  * symbolic links, and H5Oget_info_by_name always follows
116  * symbolic links. Alternatively we could use H5Lget_info
117  * and never recurse on groups discovered by symbolic
118  * links, however it could still fail if an object's
119  * reference count was manually manipulated with
120  * H5Odecr_refcount.
121  */
122  if ( group_check (od, infobuf.addr) ) {
123  printf ("%*s Warning: Loop detected!\n", spaces, "");
124  }
125  else {
126 
127  /*
128  * Initialize new operator data structure and
129  * begin recursive iteration on the discovered
130  * group. The new opdata structure is given a
131  * pointer to the current one.
132  */
133  struct opdata nextod;
134  nextod.recurs = od->recurs + 1;
135  nextod.prev = od;
136  nextod.addr = infobuf.addr;
137  return_val = H5Literate_by_name (loc_id, name, H5_INDEX_NAME,
138  H5_ITER_NATIVE, NULL, op_func, (void *) &nextod,
139  H5P_DEFAULT);
140  }
141  printf ("%*s}\n", spaces, "");
142  break;
143  case H5O_TYPE_DATASET:
144  printf ("Dataset: %s\n", name);
145  break;
146  case H5O_TYPE_NAMED_DATATYPE:
147  printf ("Datatype: %s\n", name);
148  break;
149  default:
150  printf ( "Unknown: %s\n", name);
151  }
152 
153  return return_val;
154 }

157 /************************************************************
158 
159  This function recursively searches the linked list of
160  opdata structures for one whose address matches
161  target_addr. Returns 1 if a match is found, and 0
162  otherwise.
163 
164  ************************************************************/
165 int group_check (struct opdata *od, haddr_t target_addr)
166 {
167  if (od->addr == target_addr)
168  return 1; /* Addresses match */
169  else if (!od->recurs)
170  return 0; /* Root group reached with no matches */
171  else
172  return group_check (od->prev, target_addr);
173  /* Recursively examine the next node */
174 }
