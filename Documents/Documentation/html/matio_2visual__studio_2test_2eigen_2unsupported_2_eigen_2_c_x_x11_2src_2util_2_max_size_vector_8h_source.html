58  void resize(size_t n) {
59  eigen_assert(n <= reserve_);
60  for (size_t i = size_; i < n; ++i) {
61  new (&data_[i]) T;
62  }
63  for (size_t i = n; i < size_; ++i) {
64  data_[i].~T();
65  }
66  size_ = n;
67  }
